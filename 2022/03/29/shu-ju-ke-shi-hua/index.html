//创建一个折线图
<script>
    //初始化echarts实例对象，其中参数，dom，决定了图表最终呈现的位置
    var mEchart = echarts.init(document.querySelector("div"))

    var option={
      title:{
        text:"折线"
      },
      tooltip:{
        trigger:"axis"
      },

        type:"category",
        data:{{ data1|safe }}//通过json传递从数据库中取来的数据，经过flask中render_template()传递到这
    },

        type: "value"
        },

</script>

//创建一个双柱状图
<script>

    var mEchart = echarts.init(document.querySelector('div'))
    var xData = {{data1|safe}} //通过json传递从数据库中取来的数据，经过flask中render_template()传递到这
    var yData1 = {{data2|safe}} //通过json传递从数据库中取来的数据，经过flask中render_template()传递到这
    var yData2 = {{data3|safe}} //通过json传递从数据库中取来的数据，经过flask中render_template()传递到这

    //准备配置项
    var option = {
      title:{
        text:"成绩表",
      },
      tooltip:{
        trigger:"item"
      },
      legend:{
        data:["数学成绩","语文成绩"]
      },

        type:"category",
        data:xData

            show:true

    var piedata = [
            {
                name:"淘宝",
                value:11231  
            },
            {
                name:"京东",
                value:22673
            },
            {
                name:"唯品会",
                value:6123
            },
            {
                name:"一号店",
                vlaue:8989
            },
            {
                name:"聚美优品",
                value:6700
            }
        ]
    //准备配置项
    var option={
        title:{
            text:"饼图练习",

        },
        tooltip:{
            trigger:"item",

        series:[

                label:{//饼图文字的显示
                    show : true,//默认

                        // console.log(arg.percent)

                    }

                //南丁格尔图    饼图的每个区域半径不同，随之占比增加

            }
        ]

    //将配置项设置给echarts实例对象

//创建一个散点图
<script>
  
  //初始化echarts实例对象，其中参数，dom，决定了图表最终呈现的位置
    var mChart = echarts.init(document.querySelector("div"))
	
	//此处应通过json传递从数据库中取来的数据，经过flask中render_template()传递到这，处理数据
    var data=[{"gender":"man","height":170.2,"weight":60},
    {"gender":"man","height":170.2,"weight":60},{"gender":"man","height":169.2,"weight":63},{"gender":"man","height":176,"weight":70},
    {"gender":"man","height":170,"weight":62},{"gender":"man","height":181.5,"weight":74},{"gender":"man","height":176,"weight":68}
    ]
    var axisData=[]
    for ( var i = 0; i < data.length;i++){
        var height = data[i].height
        var weight  = data[i].weight
        var newArr= [height,weight]
        axisData.push(newArr)
    }
    //console.log(axisData)

	 //准备配置项
    var option={
        title:{
                text:"散点图",
               
            },
            xAxis:{
                type:"value",
                scale:true,//拜托0值比例
            },
            yAxis:{ 
                type:"value",
                scale:true
            },
            series:{
                name:"身高体重趋势表",
                type:"scatter",
                data:axisData
            }
    }
    
    //将配置项设置给echarts实例对象
    mChart.setOption(option)
</script>

Echarst Tips

直角坐标系中的常用配置项（对于柱状图，折线图，散点图有效）
    grid：{}
        使用show属性能够将边框表现出来
        borderWidth 控制边框粗细大小
        borderColor 控制边框颜色
        通过top,left来控制图形在网页中的位置
        通过width,heigth来控制整个边框大小，边框的大小影响了整个图像的大小

坐标轴的配置：
        坐标轴类型
            type:"value"
            type:"catepory"     必须与data绑定

显示位置
           position:
                在x轴中有top和bottom两个属性
                在y轴中有left和right两个属性
